30  				direction = Direction.DOWN;
31  			}
32  		} else if (pos.x < heroPos.x) {
33  			if (pos.y > heroPos.y) {
34  				direction = Direction.DOWN;
35  			} else if (pos.y < heroPos.y) {
36  				direction = Direction.LEFT;
