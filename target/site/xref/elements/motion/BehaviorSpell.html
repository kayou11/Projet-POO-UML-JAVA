31  	
32  		if (!lorannWorld.isSpellLaunched()) {
33  			motionElement.setX(heroX);

36  		} else if (lorannWorld.isSpellLaunched() && lorannWorld.isAttractSpell()) {

64  		if ((spellX == heroX) && (spellY > heroY)) {
65  			motion.moveUp();
66  		}
67  		if ((spellX == heroX) && (spellY < heroY)) {

77  			if (Math.abs(diffX) < Math.abs(diffY)) {
78  				motion.moveDown();

106 				motion.moveRight();
107 			}
108 		}

112 			}

114 				motion.moveLeftUp();
115 			}

117 				motion.moveLeft();
